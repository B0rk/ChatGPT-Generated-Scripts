PowerShell Invoke Web Request
Invoke-WebRequest -Uri "https://<server_ip_or_hostname>/" -Method Post -InFile "C:\path\to\file.txt" -ContentType "multipart/form-data" -UseBasicParsing

PowerShell Curl
curl.exe -k -F "file=@C:\path\to\file.txt" https://<server_ip_or_hostname>/

Bash Curl
curl -k -F "file=@/path/to/file.txt" https://<server_ip_or_hostname>/
